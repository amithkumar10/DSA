
//Code for Dec Triangle
#include <iostream>
using namespace std;

int main() {
    int sum =5;
    int n = 5;
   for(int i=1; i<=n; i++){
     for(int j=i; j<=n; j++){
       
       cout<<sum<<" ";
       
    }
    cout<<endl;
    sum = sum -1;
   }

    return 0;
}
